今天经推荐学习的HTML5标签，课程老师特别有意思，最开始连“五大牛奶器”都细心讲解了一遍，然后一下子忍不住，要把H5课程上讲的标签部分看完才睡觉，课程地址：李南江-H5前端
下面是代码练习：

<!--DTD文档声明，告诉浏览器我们网页是用哪一个HTML版本规范写的-->
<!--DTD文档声明并不是一个标签-->
<!DOCTYPE html>

	<!--title 标签，指定网站的标题，还会作为用户保存网站的默认标题-->
	<title>网易云课堂极客江南H5标签学习第一天练习</title>

	<!---base（单）标签，指定全部a标签跳转的target属性->
	<!--base标签必须放在head标签之间-->
	<base target="_self">

</head>

<!--body 标签，每个.html对里只能有一对body标签>
<body>

	<!--h标签，给文本添加标题语义，共有6个，h1-h6,并且h标签里内容在网页里独占一行，实际项目开发中尽量少用h标签-->
	<h1>h标签练习</h1>

	<!--p（双）标签，告诉浏览器哪些部分是一个段落-->

	<!--hr（单）标签，在浏览器中生成一条分隔线-->
	<hr/>

	<!--br标签，表示换行-->
	<br>

	<!--img标签，告诉浏览器我们要显示图片,sur=source-->
	<!--img标签属性，width height title alt -->
	<img src="fengjing.jpg" width="300" title="这个图片是风景头像" alt="正在显示，或者显示不出来">
	<!--路径问题1.同级，直接src="name.jpg"
				2.下级，"filefoldname/name.jpg"
				3.上级，"../name.jpg"
				4.绝对路径从所在盘符开始查找-->


	<!--a（双）标签，用于指定网页界面跳转，
				格式<a href="website">作为点击的文字图像都可以</a>
				a标签href属性还可以指定一个本地地址
				还有属性title、target(_self和_blank)
				-->
	<a href="http://zhangxukun5.com" target="_self">张绪坤的博客</a>
	<a href="http://zhangxukun5.com" target="_blank" title="跳转到张绪坤的博客"><img src="fengjing.jpg" width=50></a>
	<br>
	<!--假链接#、javascript:-->
	<a href="#"># 点我会回到顶部</a>
	<a href="javascript:">javascript: 点我不会回到顶部</a>
	<br>
	<!--瞄点，通过a标签跳转到当前或者其他页面指定位置，
	那么给目标位置的标签添加一个id属性，然后指定一个独一无二的值，
	然后告诉a标签你需要跳转到的目标标签的值是什么-->
	<a href="#point">假链接跳转到point</a>
	<h2>我是point，跳转到我吧</h2>
	<!--<a href="newsite.com#point">跳到其他页面的指定位置</a>-->

	<!--<hr>标签有属性，可指定对齐方式，宽度-->
	<hr width="750">
	<!--ul标签，无序列表标签,给一堆数据添加列表语义-->
	<!--ul标签里只能有li标签，但是两者可以循环嵌套-->
	<ul>
		<li><h2>无序列表行1</h2>列表行里也有样式</li>
		<li>无序列表行2</li>
		<li>无序列表行3</li>
	</ul>

	<!--定义列表-->
	<dl>
		<dt>定义列表标题1</dt>
		<dd>定义列表内容描述1</dd>
		<dt>定义列表标题2</dt>
		<dd>定义列表内容描述2</dd>
	</dl>
	<!--定义列表可以作用于图文混排-->
	<d1>
		<dt><img src="fengjing.jpg" width="250"></dt>
	</dl>

	<!--table（双）标签，用来给一堆数据添加表格语义，当数据量比较大时，表格形式展现某中用途中会非常清晰-->
	<!--一对tr代表一行，一对td代表一行中的一个单元格-->
	<!--表格标签有边框属性border，默认边框值为0-->
	<table border="2px" bgcolor="black">
		<caption><h5>表格标题</h5></caption>
		<tr bgcolor="white">
			<td>1.1</td><td>1.2</td><td>1.3</td>
		</tr>
		<tr>
			<td>2.1</td><td>2.2</td><td>2.3</td>
		</tr>
	</table>

	<!--form（双）标签，表单标签-->
	<!--可以通过form标签中的action属性来指明提交表单内容到哪个服务器上-->
	<form action="http://www.baidu.com">
		<!--表单中常见的元素，input（单）标签，input中type属性取值不同，表单中的元素显示不同-->
		<!--label（双）标签将表单前叙述与表单框绑定-->
		<label for="account">账号：</label><input type="text" id="account">   
		密码：<input type="password">
		账号：<input type="text" value="anwen">   
		密码：<input type="password" value="anwen">
		<br>

		<!--type值为radio，代表单选框-->
		<input type="radio">不排斥单选1
		<input type="radio">不排斥单选2
		<br>

		<!--input中加name属性，值相同可以互相排斥，添加checked属性有默认-->
		<input type="radio" name="111">排斥单选1
		<input type="radio" name="111" checked="checked">加checked默认排斥单选2
		<br>

		<!--多选框checkbox-->
		<input type="checkbox">多选1
		<input type="checkbox">多选2
		<br>

		<!--重置按钮-->
		<input type="reset" value="value，值为重置">
		<!--提交按钮-->
		<input type="submit">
		<!--信息隐藏提交-->
		<input typr="hidden" name="yincan1" value="yincan2">

	</form>

	<!--type中button为普通按钮，img为图片按钮-->
	<form>
		<input type="button" value="value值，按钮">
	</form>

</body>
